        stage('git') {
            steps {

            }
        }
        stage('mvn') {
            steps {
                sh 'mvn clean package'
            }
        }
        stage('docker') {
            steps {
                sh 'docker build -t dutt1/23app:v1 .'
            }

                to: "dutt.gatiganti@gmail.com",
                from: "dutt.gatiganti@gmail.com",

            )
        }
        
    }
}
